from sqlalchemy import create_engine
from urllib.parse import quote_plus
import pandas as pd

# Connection details
username = "postgres"
password = quote_plus("**************")   # handles special characters
host = "localhost"
port = "5435"
database = "postgres"

# Create engine
engine = create_engine(
    f"postgresql+psycopg2://{username}:{password}@{host}:{port}/{database}"
)

# Create DataFrame
df = pd.DataFrame({
    "id": [1, 2, 3],
    "name": ["Amit", "Rahul", "Sneha"]
})

# Export to PostgreSQL
df.to_sql(
    "employee_test",
    con=engine,
    schema="public",
    if_exists="replace",
    index=False
)

print("Data exported successfully")
